    # 1 Наследование на ES6

    Условие

    • Реализовать
    ⚡️ Реализация 
    class User {
    constructor(name) {
        this.name = name;
    }
    getName() {
        console.log(this.name);
    }
}

class Admin extends User {
    constructor(name) {
        super(name);
    }
}

const admin = new Admin('Ivan');
admin.getName();    // => "Ivan"
console.log(admin); // => Object { name: "Ivan" }
