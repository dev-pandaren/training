    # Всплытие

    # Теория

    • Двухпроходной анализатор
    • Всплытие (всплытие для var, let, const, function)
    • Temporal Dead Zone

    # Задачи

    # 1 Всплытие

    Условие

    let a = 1;

function b() {
    a = 10;
    return 20;
    function a() {}
}

b();

console.log(a);
    ⚡️ Реализация 
    let a = 1;

function b() {
    // перезаписывается не глобальная переменная, а локальная функция
    a = 10;
    return 20;
    // функция всплывает
    function a() {}
}

b();

console.log(a); // 1

    # 2 Всплытие

    Условие

    function a() { return true }
var a;
console.log(a);

    function a() { return true }
var a = 'Hello';
console.log(a);
    ⚡️ Реализация
    • При обращении до объявления: вызывается функция
    • Функция имеет безусловный приоритет
    // Переменная "var a" непроинициализирована
function a() { return true }
var a;
console.log(a); //=> ƒ a() { return true }

    // Переменная "var a" проинициализирована
function a() { return true }
var a = 'Hello';
console.log(a); // => "Hello"

    # 3 Значение массива

    Условие

    var a = [];

function clear(arr) {
    arr.push(2);
    arr = null;
}

clear(a);

console.log(a);
    ⚡️ Реализация 
    const a = [];

function clear(arr) {
    // присвоение в локальную переменную
    arr.push(2);
    arr = null;
}
clear(a);

console.log(a); // [2]

    # 4 Всплытие

    Условие

    function foo(a) {  
    if (a > 0) {
        let a = a + 10;
        return a;
    }
    return a;
}

console.log(foo(20));
    ⚡️ Реализация 
    function foo(a) {  
    if (a > 0) {
        // a во временной dead zone
        let a = a + 10;
        return a;
    }
    return a;
}

console.log(foo(20));

// => ReferenceError: Cannot access 'a' before initialization

    # 5 Переменная

    Условие

    var a = 5;
function someFunction() {
    a = 3;

    if (false) {
        var a = 10;
    }
}

someFunction();

console.log(a);
    ⚡️ Реализация 
    var a = 5;
function someFunction() {
    // "3" запишется не в глобальную, а в локальную, потому что она всплыла
    a = 3;

    // условие не выполнится но "a" всплывет
    if (false) {
        var a = 10;
    }
}

someFunction();
console.log(a); // => 5
