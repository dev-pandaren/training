    # CSS селекторы (>, +, ~)

    ul li > span   /* не все span, а только дочерние span */
ul li + li     /* все последующие элементы li */
.class1 ~ div  /* все div элементы после class1 */
.class1 ~ *    /* все элементы после class1 */

    <div class="container">
	<div>item</div>
	<div>item</div>
	<div>
		item
		<div>inside item</div>
		<div>inside item</div>
	</div>
</div>

<div>outside</div>
<div>outside</div>
<p>outside</p>
