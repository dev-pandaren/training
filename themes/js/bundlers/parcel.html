+ public
	- favicon.png
	- index.html
+ src
	- index.js
- .gitignore
- package.json

# index.html

<!DOCTYPE html>
<html>
<head>
	...
</head>
<body>
	<script src="../src/index.js"></script>
</body>
</html>

# package.json

// package.json для Parcel
{
	"name": "app",
	"version": "1.0.0",
	"scripts": {
		"start": "parcel public/index.html --open",
		"build": "parcel build public/index.html"
	},
	"dependencies": {
		"axios": "^0.19.2"
	},
	"devDependencies": {
		"parcel-bundler": "^1.12.4"
	}
}
