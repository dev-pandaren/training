    # Объекты

    const user = {
	name: 'Tony',
	email: 'tony@mail.com',
	age: 25,
	address: {
		street: '555 Some st',
		city: 'Boston'
	},
	getData() {
		return this.name + ' ' + this.email;
	}
}

console.log(user.name);
console.log(user.address.street);
console.log(user.getData());

    # Цикл for..in для перебора ключей объекта

    for(let key in obj) {
	console.log(key);      // ключи 
	console.log(obj[key]); // значения
}
